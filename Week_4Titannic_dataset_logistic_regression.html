def sigmoid(h):
  return 1/(1+np.exp(-h))

def cross_entropy(y,p_hat): #this is binary cross entropy
  return -(1/len(y))*np.sum(y*np.log(p_hat)+(1-y)*np.log(1-p_hat))

def accuracy(y,y_hat):
  return np.mean(y==y_hat)

def normalize_features(X):
    return (X - X.mean(axis=0)) / X.std(axis=0)

class SLogisticRegression():

  def __init__(self,thresh = 0.5):
    self.thresh = thresh 
    self.b = None
    self.w = None

  def fit(self,X,y,eta=1e-3,epochs=1e3,show_curve=False):
    epochs = int(epochs)
    N,D = X.shape
    # initialize the weight and biases
    self.w = np.random.randn(D)
    self.b = np.random.randn(1)

    J=np.zeros(epochs)
    #SDG Optimization
    for epoch in range(epochs):
      p_hat = self.__forward__(X)
      J[epoch] = cross_entropy(y,p_hat)
      #weight and biases updated rules
      self.w -= eta*(1/N)*X.T@(p_hat-y)
      self.b -= eta*(1/N)*np.sum(p_hat-y)

    if show_curve:
      plt.figure()
      plt.plot(J)
      plt.xlabel("epochs")
      plt.ylabel("$\mathcal{J}$")
      plt.title("Training Curve")

  def __forward__(self,X):
    return sigmoid(X@self.w + self.b)

  def predict(self,X):
    return(self.__forward__(X)>=self.thresh).astype(np.int32)

def preprocess_data(data):
    # Feature Engineering
    data['Age_bin'] = pd.cut(data['Age'], bins=[0, 12, 20, 40, 60, 100], labels=False)
    data['Fare_bin'] = pd.qcut(data['Fare'], q=4, labels=False)
    data['FamilySize_bin'] = pd.cut(data['FamilySize'], bins=[0, 1, 4, 20], labels=False)
    
    # Select Features
    features = ['Pclass_1', 'Pclass_2', 'Pclass_3', 'Sex', 'Age_bin', 'Fare_bin',
                'Title_Master', 'Title_Miss', 'Title_Mr', 'Title_Mrs', 'Title_Officer', 'Title_Royalty',
                'Embarked_C', 'Embarked_Q', 'Embarked_S', 'FamilySize_bin']

    return data[features], data['Survived']

def train_test_split(preprocess_data, test_size=0.2):
    num_test = int(test_size * len(preprocess_data))
    num_train = len(preprocess_data) - num_test
    train_data = preprocess_data[:num_train].reset_index(drop=True)
    test_data = preprocess_data[num_train:].reset_index(drop=True)
    return train_data, test_data

def softmax(h):
    return (np.exp(h).T / np.sum(np.exp(h), axis=1)).T

def cross_entropy(Y, P_hat):
    return -(1 / len(Y)) * np.sum(np.sum(Y * np.log(P_hat), axis=1), axis=0)

def indices_to_one_hot(data,nb_classes):
  target = np.array(data).reshape(-1)
  return np.eye(nb_classes)[target]

def accuracy(y,y_hat):
  return np.mean(y==y_hat)

class MVlogisticRegression():

    def fit(self, X, y, eta=1e-3, epochs=100, show_curve=False):
        epochs = int(epochs)
        N, D = X.shape
        K = len(np.unique(y))
        y_values = np.unique(y, return_inverse=True)[1]
        Y = indices_to_one_hot(y_values, K).astype(int)
        # Initialize W,B
        self.B = np.random.randn(1, K)
        self.W = np.random.randn(D, K)

        J = np.zeros(epochs)

        for epoch in range(epochs):
            P_hat = self.__forward__(X)
            J[epoch] = cross_entropy(Y, P_hat)
            # Weight and Bias updates rules
            self.W -= eta * (1 / N) * X.T @ (P_hat - Y)
            self.B -= eta * (1 / N) * np.sum(P_hat - Y, axis=0)

        if show_curve:
            plt.figure()
            plt.plot(J)
            plt.xlabel("epochs")
            plt.ylabel("$\mathcal{J}$")
            plt.title("Training Curve")

    def __forward__(self, X):
        return softmax(X @ self.W + self.B)

    def predict(self, X):
        return np.argmax(self.__forward__(X), axis=1)

def predict_survival(log_reg, pclass_1, pclass_2, pclass_3, sex, age_bin, fare_bin, title_master, title_miss, title_mr, title_mrs, title_officer, title_royalty, embarked_c, embarked_q, embarked_s, family_size_bin):
    # Prepare the input data
    input_data = pd.DataFrame({'Pclass_1': [pclass_1], 'Pclass_2': [pclass_2], 'Pclass_3': [pclass_3], 'Sex': [sex], 'Age_bin': [age_bin], 'Fare_bin': [fare_bin],
                'Title_Master': [title_master], 'Title_Miss': [title_miss], 'Title_Mr': [title_mr], 'Title_Mrs': [title_mrs], 'Title_Officer': [title_officer], 'Title_Royalty': [title_royalty],
                'Embarked_C': [embarked_c], 'Embarked_Q': [embarked_q], 'Embarked_S': [embarked_s], 'FamilySize_bin': [family_size_bin]})
    
    # Make predictions
    prediction = log_reg.predict(input_data)
    
    # Return the prediction result
    return "Survived" if prediction[0] == 1 else "Did not survive"
